# 3D Home View Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement task-by-task. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Add an opt-in, mobile-friendly 3D "Home" view where users place plants in up to 2 of 5 cozy low-poly rooms and tap them to open the existing detail sheet; plants droop when overdue and perk up when watered.

**Architecture:** Procedural react-three-fiber scene, lazy-loaded (no SSR), toggled against the existing grid inside `GardenClient`. Data: `users.home_spaces` (≤2 room keys) + `plants.room`. Reuses `PlantDetailSheet`, watering API, and `computeCountdown`.

**Tech Stack:** three 0.185, @react-three/fiber 9, @react-three/drei 10, Next 15, Postgres/RLS.

### Task 1: Room catalog + validators (`src/lib/home.ts`)

**Files:** Create `src/lib/home.ts`; Test inline via tsx script.

- [ ] **Step 1:** Write `home.ts`:

```ts
export const ROOM_KEYS = [
  "living_room", "kitchen", "bedroom", "bathroom", "balcony",
] as const;
export type RoomKey = (typeof ROOM_KEYS)[number];

export const ROOMS: Record<RoomKey, { label: string; blurb: string }> = {
  living_room: { label: "Living room", blurb: "Bright, sociable, medium light" },
  kitchen: { label: "Kitchen", blurb: "Warm and steamy by the window" },
  bedroom: { label: "Bedroom", blurb: "Calm, lower light" },
  bathroom: { label: "Bathroom", blurb: "Humid, low light — ferns love it" },
  balcony: { label: "Balcony", blurb: "The brightest spot you've got" },
};

export function isRoomKey(v: unknown): v is RoomKey {
  return typeof v === "string" && (ROOM_KEYS as readonly string[]).includes(v);
}

/** Validate a proposed home_spaces array: ≤2, unique, all valid keys. */
export function normalizeHomeSpaces(v: unknown): RoomKey[] | null {
  if (!Array.isArray(v)) return null;
  const out: RoomKey[] = [];
  for (const item of v) {
    if (!isRoomKey(item)) return null;
    if (!out.includes(item)) out.push(item);
  }
  if (out.length > 2) return null;
  return out;
}

/** Local-space slot positions [x,y,z] per room (pot base sits on the surface). */
export const SLOTS: Record<RoomKey, [number, number, number][]> = {
  living_room: [[-1.1,0.62,-0.9],[ -0.4,0.62,-0.9],[1.0,0.0,0.4],[-1.2,0.0,0.9],[0.3,0.0,1.0]],
  kitchen: [[-1.1,0.92,-0.9],[-0.4,0.92,-0.9],[0.3,0.92,-0.9],[1.1,0.92,-0.5],[1.0,0.0,0.8]],
  bedroom: [[-1.2,0.55,-0.8],[1.1,0.0,-0.9],[-1.2,0.0,0.9],[0.4,0.0,1.0],[1.1,0.0,0.6]],
  bathroom: [[1.0,0.62,-0.8],[-1.2,0.9,-0.9],[-1.1,0.0,0.9],[0.2,0.0,1.0],[1.0,0.0,0.7]],
  balcony: [[-1.1,0.5,-0.9],[-0.3,0.5,-0.9],[0.5,0.5,-0.9],[-1.0,0.0,0.8],[0.6,0.0,0.9]],
};

export const SLOTS_PER_ROOM = 5;
```
